27namespace uniot {
28class NetworkController : public ISchedulerConnectionKit, public CoreEventListener, public CBORStorage {
29 public:
30 enum Topic { NETWORK_LED = FOURCC(nled) };
31
32 NetworkController(NetworkScheduler &network,
33 uint8_t pinBtn = UINT8_MAX,
34 uint8_t activeLevelBtn = LOW,
35 uint8_t pinLed = UINT8_MAX,
36 uint8_t activeLevelLed = HIGH,
37 uint8_t maxRebootCount = 3,
38 uint32_t rebootWindowMs = 10000)
39 : CBORStorage("ctrl.cbor"),
40 mpNetwork(&network),
41 mNetworkLastState(NetworkScheduler::SUCCESS),
42 mClickCounter(0),
43 mPinLed(pinLed),
44 mActiveLevelLed(activeLevelLed),
45 mMaxRebootCount(maxRebootCount),
46 mRebootWindowMs(rebootWindowMs),
47 mRebootCount(0) {
48 if (pinLed != UINT8_MAX) {
49 pinMode(mPinLed, OUTPUT);
50 }
51 if (pinBtn != UINT8_MAX) {
52 mpConfigBtn = MakeUnique<Button>(pinBtn, activeLevelBtn, 30, [&](Button *btn, Button::Event event) {
53 switch (event) {
54 case Button::LONG_PRESS:
55 if (mClickCounter > 3)
56 mpNetwork->forget();
57 else
58 mpNetwork->reconnect();
59 break;
60 case Button::CLICK:
61 if (!mClickCounter) {
62 mpTaskResetClickCounter->attach(5000, 1);
63 }
64 mClickCounter++;
65 default:
66 break;
67 }
68 });
69 }
70 _initTasks();
71 _checkAndHandleReboot();
72 CoreEventListener::listenToEvent(NetworkScheduler::Topic::CONNECTION);
73 }
74
75 virtual ~NetworkController() {
76 CoreEventListener::stopListeningToEvent(NetworkScheduler::Topic::CONNECTION);
77 mpNetwork = nullptr;
78 }
79
80 virtual bool store() override {
81 object().put("reset", mRebootCount);
82 return CBORStorage::store();
83 }
84
85 virtual bool restore() override {
86 if (CBORStorage::restore()) {
87 mRebootCount = object().getInt("reset");
88 return true;
89 }
90 return false;
91 }
92
93 virtual void onEventReceived(unsigned int topic, int msg) override {
94 if (NetworkScheduler::CONNECTION == topic) {
95 int lastState = _resetNetworkLastState(msg);
96 switch (msg) {
97 case NetworkScheduler::ACCESS_POINT:
98 if (lastState != NetworkScheduler::FAILED) {
99 statusWaiting();
100 }
101 break;
102 case NetworkScheduler::SUCCESS:
103 statusIdle();
104 break;
105 case NetworkScheduler::CONNECTING:
106 statusBusy();
107 break;
108 case NetworkScheduler::DISCONNECTED:
109 if (lastState != NetworkScheduler::CONNECTING) {
110 // If previous state was CONNECTING, most likely there was a manual reconnect request
111 mpNetwork->reconnect();
112 }
113 break;
114 case NetworkScheduler::FAILED:
115 default:
116 statusAlarm();
117 break;
118 }
119 }
120 }
121
122 virtual void pushTo(TaskScheduler &scheduler) override {
123 scheduler.push("signal_led", mpTaskSignalLed);
124 scheduler.push("rst_reboot_count", mpTaskResetRebootCounter);
125 if (_hasButton()) {
126 scheduler.push("btn_config", mpTaskConfigBtn);
127 scheduler.push("rst_click_count", mpTaskResetClickCounter);
128 }
129 }
130
131 virtual void attach() override {
132 mpTaskResetRebootCounter->once(mRebootWindowMs);
133 if (_hasButton()) {
134 mpTaskConfigBtn->attach(100);
135 }
136 statusBusy();
137 }
138
139 void statusWaiting() {
140 mpTaskSignalLed->attach(1000);
141 }
142
143 void statusBusy() {
144 mpTaskSignalLed->attach(500);
145 }
146
147 void statusAlarm() {
148 mpTaskSignalLed->attach(200);
149 }
150
151 void statusIdle() {
152 mpTaskSignalLed->attach(200, 1);
153 }
154
155 Button *getButton() {
156 return _hasButton() ? mpConfigBtn.get() : nullptr;
157 }
158
159 private:
160 void _initTasks() {
161 mpTaskSignalLed = TaskScheduler::make([&](SchedulerTask &self, short t) {
162 static bool signalLevel = true;
163 signalLevel = (!signalLevel && t);
164 CoreEventListener::emitEvent(Topic::NETWORK_LED, signalLevel);
165
166 if (_hasLed()) {
167 digitalWrite(mPinLed, signalLevel ? mActiveLevelLed : !mActiveLevelLed);
168 }
169 });
170
171 if (_hasButton()) {
172 mpTaskConfigBtn = TaskScheduler::make(*mpConfigBtn);
173 mpTaskResetClickCounter = TaskScheduler::make([&](SchedulerTask &self, short t) {
174 UNIOT_LOG_DEBUG("ClickCounter = %d", mClickCounter);
175 mClickCounter = 0;
176 });
177 }
178
179 mpTaskResetRebootCounter = TaskScheduler::make([&](SchedulerTask &self, short t) {
180 mRebootCount = 0;
181 NetworkController::store();
182 });
183 }
184
185 void _checkAndHandleReboot() {
186 NetworkController::restore();
187 mRebootCount++;
188 if (mRebootCount >= mMaxRebootCount) {
189 mpTaskResetRebootCounter->detach();
190 mpNetwork->forget();
191 mRebootCount = 0;
192 }
193 NetworkController::store();
194 }
195
196 int _resetNetworkLastState(int newState) {
197 auto oldState = mNetworkLastState;
198 mNetworkLastState = newState;
199 return oldState;
200 }
201
202 inline bool _hasButton() {
203 return mpConfigBtn.get() != nullptr;
204 }
205
206 inline bool _hasLed() {
207 return mPinLed != UINT8_MAX;
208 }
209
210 NetworkScheduler *mpNetwork;
211 int mNetworkLastState;
212
213 uint8_t mClickCounter;
214 uint8_t mPinLed;
215 uint8_t mActiveLevelLed;
216 uint8_t mMaxRebootCount;
217 uint32_t mRebootWindowMs;
218 uint8_t mRebootCount;
219
220 UniquePointer<Button> mpConfigBtn;
221
222 TaskScheduler::TaskPtr mpTaskSignalLed;
223 TaskScheduler::TaskPtr mpTaskConfigBtn;
224 TaskScheduler::TaskPtr mpTaskResetClickCounter;
225 TaskScheduler::TaskPtr mpTaskResetRebootCounter;
226};
227} // namespace uniot
