47Object dwrite(Root root, VarObject env, VarObject list) {
48 auto expeditor = PrimitiveExpeditor::describe(name::dwrite, Lisp::Bool, 2, Lisp::Int, Lisp::BoolInt)
49 .init(root, env, list);
50 expeditor.assertDescribedArgs();
51 auto pin = expeditor.getArgInt(0);
52 auto state = expeditor.getArgBool(1);
53
54 uint8_t gpio = 0;
55 if (pin >= 0 && expeditor.getAssignedRegister().getGpio(pin, gpio)) {
56 digitalWrite(gpio, state);
57 } else {
58 expeditor.terminate("pin is out of range");
59 }
60
61 return expeditor.makeBool(state);
62}

169 .init(root, env, list);

173

175
176 if (btn) {
177 value = btn->resetClick();
178 } else {
179 expeditor.terminate("wrong button id");
180 }
181
