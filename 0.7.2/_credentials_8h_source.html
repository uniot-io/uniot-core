192 uint8_t signature[64];

216 String _calcDeviceId() {
217 uint8_t mac[6];
218 char macStr[13] = {0};
219#if defined(ESP8266)
220 wifi_get_macaddr(STATION_IF, mac);
221#elif defined(ESP32)
222 esp_read_mac(mac, ESP_MAC_WIFI_STA);
223#endif
224 for (uint8_t i = 0; i < 6; i++)
225 sprintf(macStr + i * 2, "%02x", mac[i]);
226
227 return String(macStr);
228 }

235 void _generatePrivateKey() {
236 // Initialize the random number generator with device-specific entropy
237 RNG.begin(String("uniot::entropy::" + mCreatorId + "::" + mDeviceId).c_str());
238
239 uint8_t privateKey[32];
240 Ed25519::generatePrivateKey(privateKey);
241 mPrivateKey = Bytes(privateKey, sizeof(privateKey));
242 }
243
249 void _derivePublicKey() {
250 uint8_t publicKey[32];
251 Ed25519::derivePublicKey(publicKey, mPrivateKey.raw());
252 mPublicKeyRaw = Bytes(publicKey, sizeof(publicKey));
253 mPublicKey = mPublicKeyRaw.toHexString();
254 }
