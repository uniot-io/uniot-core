707 Bytes _build(cn_cbor *cb, bool visitSiblings = true) const {
708 auto calculated = cn_cbor_encoder_write(NULL, 0, 0, cb, visitSiblings);
709 Bytes bytes(nullptr, calculated);
710 auto written = bytes.fill([&](uint8_t *buf, size_t size) {
711 auto actual = cn_cbor_encoder_write(buf, 0, size, cb, visitSiblings);
712 if (actual < 0) {
713 UNIOT_LOG_ERROR("%s", "CBORObject build failed, buffer size too small");
714 return 0;
715 }
716 return actual;
717 });
718
719 return bytes.prune(written);
720 }

809 String _getValueAsString(cn_cbor *cb) const {
810 // if(!cb) throw "error"; // TODO: ???

812 if (CN_CBOR_TEXT == cb->type) {
813 auto bytes = Bytes(cb->v.bytes, cb->length);
814 bytes.terminate();
815 return String(bytes.c_str());
816 }
817 if (CN_CBOR_INT == cb->type) {
818 return String(cb->v.sint);

820 if (CN_CBOR_UINT == cb->type) {
821 return String(cb->v.uint);
822 }
823 if (CN_CBOR_FLOAT == cb->type) {
824 return String(cb->v.f);
825 }
826 if (CN_CBOR_DOUBLE == cb->type) {
827 return String(cb->v.dbl);
828 }
829 if (CN_CBOR_TRUE == cb->type) {
830 return String("1");
831 }
832 if (CN_CBOR_FALSE == cb->type) {
833 return String("0");
834 }
835 }
836 return "";
