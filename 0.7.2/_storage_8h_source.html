60class Storage {
61 public:
71 Storage(const String &path) {
72 setPath("/" + path);
73
74 sInstancesCount++;
75
76 if (!sMounted) {
77#if defined(ESP8266)
78 sMounted = FileFS.begin();
79#elif defined(ESP32)
80 sMounted = FileFS.begin(true);
81#endif
82 // By default, FileFS will automatically format the filesystem if one is not detected.
83 // This means that you will lose all filesystem data even if other filesystem exists.
84 UNIOT_LOG_WARN_IF(!sMounted, "Failed to mount the file system");
85 }
86 }
87
94 virtual ~Storage() {
95 sInstancesCount--;
96 if (!sInstancesCount) {
97 unmount();
98 }
99 }
100
111 static void unmount() {
112 if (sMounted) {
113 FileFS.end();
114 sMounted = false;
115 }
116 }
117
128 virtual bool store() {
129 auto file = FileFS.open(mPath, "w");
130 if (!file) {
131 UNIOT_LOG_WARN("Failed to open %s", mPath.c_str());
132 return false;
133 }
134 file.write(mData.raw(), mData.size());
135 file.close();
136
137#if defined(ESP8266)
138#if UNIOT_USE_LITTLEFS != 1
139 // using SPIFFS.gc() can avoid or reduce issues where SPIFFS reports
140 // free space but is unable to write additional data to a file
141 UNIOT_LOG_DEBUG_IF(!FileFS.gc(), "SPIFFS gc failed. That's all right. Caller: %s", mPath.c_str());
142#endif
143#endif
144
145 return true;
146 }
147
158 virtual bool restore() {
159 auto file = FileFS.open(mPath, "r");
160 if (!file) {
161 UNIOT_LOG_WARN("Failed to open %s. It is ok on first start", mPath.c_str());
162 return false;
163 }
164 mData = _readSmallFile(file); // TODO: cleanup before read?
165 file.close();
166 return true;
167 }
168
178 virtual bool clean() {
179 mData.clean();
180 if (!FileFS.remove(mPath)) {
181 UNIOT_LOG_WARN("Failed to remove %s", mPath.c_str());
182 return false;
183 }
184 return true;
185 }
186
187 protected:
194 Bytes mData;
195
201 String mPath;
202
208 void setPath(const String &path) {
209 mPath = path;
210 // There is a limit of 32 chars in total for filenames.
211 // One '\0' char is reserved for string termination.
212 UNIOT_LOG_WARN_IF(path.length() > 31, "Path length of '%s' > 31 chars", mPath.c_str());
213 }
214
215 private:
225 Bytes _readSmallFile(File &file) {
226 auto toRead = file.size() - file.position();
227 char *buf = (char *)malloc(toRead);
228 auto countRead = file.readBytes(buf, toRead);
229 const Bytes data((uint8_t *)buf, countRead);
230 free(buf);
231 return data;
232 }
