42 class NetworkScheduler : public ISchedulerConnectionKit, public CoreEventEmitter
43 {
44 public:
45 enum Channel { OUT_SSID = FOURCC(ssid) };
46 enum Topic { CONNECTION = FOURCC(netw) };
47 enum Msg { FAILED = 0, SUCCESS, CONNECTING, DISCONNECTED, ACCESS_POINT };
48
49 NetworkScheduler(Credentials &credentials)
50 : mpCredentials(&credentials),
51 mApSubnet(255, 255, 255, 0),
52 mConfigServer(IPAddress(1, 1, 1, 1))
53 {
54 mApName = "UNIOT-" + String(mpCredentials->getShortDeviceId(), HEX);
55 mApName.toUpperCase();
56
57 // default wifi persistent storage brings unexpected behavior, I turn it off
58 WiFi.persistent(false);
59 WiFi.setAutoConnect(false);
60 WiFi.setAutoReconnect(false);
61
62 WiFi.mode(WIFI_STA);
63 _initTasks();
64 _initServerCallbacks();
65 }
66
67 virtual void pushTo(TaskScheduler &scheduler) override {
68 scheduler.push("server_start", mTaskStart);
69 scheduler.push("server_serve", mTaskServe);
70 scheduler.push("server_stop", mTaskStop);
71 scheduler.push("ap_config", mTaskConfigAp);
72 scheduler.push("sta_connect", mTaskConnectSta);
73 scheduler.push("sta_connecting", mTaskConnecting);
74 scheduler.push("wifi_monitor", mTaskMonitoring);
75 }
76
77 virtual void attach() override {
78 mWifiStorage.restore();
79 if(mWifiStorage.isCredentialsValid()) {
80 mTaskConnectSta->attach(500, 1);
81 } else {
82 mTaskConfigAp->attach(500, 1);
83 }
84 }
85
86 void forget() {
87 mWifiStorage.clean();
88 mTaskConfigAp->attach(500, 1);
89 }
90
91 bool reconnect() {
92 if(mWifiStorage.isCredentialsValid()) {
93 mTaskConfigAp->detach();
94 mTaskConnectSta->attach(500, 1);
95 return true;
96 }
97 return false;
98 }

100 private:
101 void _initTasks() {
102 mTaskStart = TaskScheduler::make([this](SchedulerTask &self, short t) {
103 if(mConfigServer.start()) {
104 mTaskServe->attach(10);
105 }
106 });
107 mTaskServe = TaskScheduler::make(mConfigServer);
108 mTaskStop = TaskScheduler::make([this](SchedulerTask &self, short t) {
109 mTaskServe->detach();
110 mConfigServer.stop();
111 });
112
113 mTaskConfigAp = TaskScheduler::make([this](SchedulerTask &self, short t) {
114 WiFi.disconnect(true);
115 WiFi.softAPdisconnect(true);
116 if( WiFi.softAPConfig(mConfigServer.ip(), mConfigServer.ip(), mApSubnet)
117 && WiFi.softAP(mApName.c_str()))
118 {
119#if defined(ESP32) && defined(ENABLE_LOWER_WIFI_TX_POWER)
120 WiFi.setTxPower(WIFI_TX_POWER_LEVEL);
121#endif
122 mTaskStart->attach(500, 1);
123 CoreEventEmitter::sendDataToChannel(Channel::OUT_SSID, Bytes(mApName));
124 CoreEventEmitter::emitEvent(Topic::CONNECTION, Msg::ACCESS_POINT);
125 } else {
126 UNIOT_LOG_WARN("NetworkScheduler, mTaskConfigAp failed");
127 mTaskConfigAp->attach(500, 1);
128 }
129 });
130 mTaskConnectSta = TaskScheduler::make([this](SchedulerTask &self, short t) {
131 WiFi.disconnect(true, true);
132 bool disconect = true;
133 WiFi.softAPdisconnect(true); // !!!!!!!!!!!!
134 WiFi.setHostname(mApName.c_str());
135 bool connect = WiFi.begin(mWifiStorage.getSsid().c_str(), mWifiStorage.getPassword().c_str()) != WL_CONNECT_FAILED;
136 if (disconect && connect)
137 {
138#if defined(ESP32) && defined(ENABLE_LOWER_WIFI_TX_POWER)
139 WiFi.setTxPower(WIFI_TX_POWER_LEVEL);
140#endif
141 mTaskServe->detach();
142 mTaskConnecting->attach(100, 100);
143 CoreEventEmitter::sendDataToChannel(Channel::OUT_SSID, Bytes(mWifiStorage.getSsid()));
144 CoreEventEmitter::emitEvent(Topic::CONNECTION, Msg::CONNECTING);
145 }
146 });
147 mTaskConnecting = TaskScheduler::make([this](SchedulerTask &self, short times) {
148 auto __processFailure = [this]() {
149 const int triesBeforeGivingUp = 3;
150 static int tries = 0;
151 if(++tries < triesBeforeGivingUp) {
152 UNIOT_LOG_INFO("Tries to connect until give up is %d", triesBeforeGivingUp - tries);
153 mTaskConnectSta->attach(500, 1);
154 } else {
155 tries = 0;
156 mTaskConfigAp->attach(500, 1);
157 CoreEventEmitter::emitEvent(Topic::CONNECTION, Msg::FAILED);
158 }
159 };
160
161 switch(WiFi.status()){
162 case WL_CONNECTED:
163 mTaskConnecting->detach();
164 mTaskStop->attach(500, 1);
165 mTaskMonitoring->attach(2000);
166 mWifiStorage.store();
167 mpCredentials->store(); // NOTE: it is stored each time it is connected to the network
168 CoreEventEmitter::emitEvent(Topic::CONNECTION, Msg::SUCCESS);
169 break;
170
171 case WL_NO_SSID_AVAIL:
172 case WL_CONNECT_FAILED:
173 mTaskConnecting->detach();
174 __processFailure();
175 break;
176
177 case WL_DISCONNECTED:
178 if(!times) {
179 __processFailure();
180 }
181 break;
182
183 case WL_IDLE_STATUS:
184 default:
185 break;
186 }
187 });
188 mTaskMonitoring = TaskScheduler::make([this](SchedulerTask &self, short times) {
189 if(WiFi.status() != WL_CONNECTED) {
190 mTaskMonitoring->detach();
191 CoreEventEmitter::emitEvent(Topic::CONNECTION, Msg::DISCONNECTED);
192 }
193 });
194 }

196 void _initServerCallbacks() {
197 mConfigServer.get()->onNotFound([this] {
198 mConfigServer.get()->sendHeader("Location", "/", true);
199 mConfigServer.get()->send_P(307, text, text);
200 });
201
202 mConfigServer.get()->on("/", [this] {
203 mConfigServer.get()->sendHeader("Content-Encoding", "gzip");
204 mConfigServer.get()->send_P(200, text_html, CONFIG_MIN_HTML_GZ, CONFIG_MIN_HTML_GZ_LENGTH);
205 });
206
207 mConfigServer.get()->on("/scan", [this] {
208 // TODO: implement async scan
209 auto n = WiFi.scanNetworks();
210 String networks = "[";
211 for (auto i = 0; i < n; ++i) {
212 networks += '"' + WiFi.SSID(i) + '"';
213 if (i < n - 1)
214 networks += ',';
215 }
216 networks += ']';
217 UNIOT_LOG_DEBUG("Networks: %s", networks.c_str());
218 mConfigServer.get()->send(200, text, networks);
219 });
220
221 mConfigServer.get()->on("/config", [this] {
222 mWifiStorage.setCredentials(mConfigServer.get()->arg("ssid"), mConfigServer.get()->arg("pass"));
223 mConfigServer.get()->sendHeader("Location", "/", true);
224 mConfigServer.get()->send_P(307, text, text);
225 if(mWifiStorage.isCredentialsValid()) {
226 mTaskConnectSta->attach(500, 1);
227 mpCredentials->setOwnerId(mConfigServer.get()->arg("acc"));
228 }
229 });
230 }
