105 bool store() {
106 object()
107 .put("persist", mPersist)
108 .put("checksum", (int)mChecksum)
109 .put("code", mPersist ? getLisp().getLastCode().c_str() : "")
110 .forceDirty(); // Avoid optimization. The pointer to the data may be the same. Read the warnings.

124 virtual void onEventReceived(unsigned int topic, int msg) override {
125 if (topic == unLisp::Topic::OUT_LISP_MSG) {
126 if (msg == unLisp::Msg::OUT_MSG_ERROR) {
127 CoreEventListener::receiveDataFromChannel(unLisp::Channel::OUT_LISP_ERR, [this](unsigned int id, bool empty, Bytes data) {
128 if (!empty) {
129 mFailedWithError = true;
130
131 CBORObject packet;
132 packet.put("type", "error");
133 packet.put("timestamp", static_cast<int64_t>(Date::now()));
134 packet.put("msg", data.c_str());
135 publishDevice("debug/err", packet.build(), true);
136 UNIOT_LOG_ERROR("lisp error: %s", data.c_str());
137 }
138 });
139 return;
140 }
141 if (msg == unLisp::Msg::OUT_MSG_LOG) {
142 CoreEventListener::receiveDataFromChannel(unLisp::Channel::OUT_LISP_LOG, [this](unsigned int id, bool empty, Bytes data) {
143 if (!empty) {
144 CBORObject packet;
145 packet.put("type", "log");
146 packet.put("timestamp", static_cast<int64_t>(Date::now()));
147 packet.put("msg", data.c_str());
148 publishDevice("debug/log", packet.build());
149 UNIOT_LOG_INFO("lisp log: %s", data.c_str());
150 }
151 });
152 return;
153 }
154 if (msg == unLisp::Msg::OUT_MSG_ADDED) {
155 CoreEventListener::receiveDataFromChannel(unLisp::Channel::OUT_LISP, [](unsigned int id, bool empty, Bytes data) {
156 // NOTE: it is currently only used for debugging purposes
157 UNIOT_LOG_TRACE_IF(!empty, "lisp: %s", data.toString().c_str());
158 });
159 return;
160 }
161 return;
162 }
163 if (topic == unLisp::Topic::OUT_LISP_REQUEST) {
164 if (msg == unLisp::Msg::OUT_REFRESH_EVENTS) {
165 // This is necessary to retrieve events marked as `retained` during the execution of a new script
166 this->unsubscribe(mTopicEvents);
167 this->subscribe(mTopicEvents);
168 return;
169 }
170 return;
171 }
172 if (topic == unLisp::Topic::OUT_LISP_EVENT) {
173 if(msg == unLisp::Msg::OUT_NEW_EVENT) {
174 CoreEventListener::receiveDataFromChannel(unLisp::Channel::OUT_EVENT, [this](unsigned int id, bool empty, Bytes data) {
175 if (!empty) {
176 CBORObject event(data);
177 event.put("timestamp", static_cast<int64_t>(Date::now()))
178 .putMap("sender")
179 .put("type", "device")
180 .put("id", MQTTDevice::getDeviceId().c_str());
181 auto eventData = event.build();
182 auto eventID = event.getString("eventID");
183 MQTTDevice::publishGroup("all", "event/" + eventID, eventData);
184 }
185 });
186 }
187 return;
188 }
189 }

199 virtual void handle(const String &topic, const Bytes &payload) override {
200 if (MQTTDevice::isTopicMatch(mTopicScript, topic)) {
201 MQTTDevice::publishEmptyDevice("debug/err"); // clear previous errors
202 handleScript(payload);
203 return;
204 }
205 if (MQTTDevice::isTopicMatch(mTopicEvents, topic)) {
206 handleEvent(payload);
207 return;
208 }
209 }
210
220 void handleScript(const Bytes &payload) {
221 static bool firstPacketReceived = false;
222 CBORObject packet(payload);
223 auto script = Bytes(packet.getString("code"));
224 auto newPersist = packet.getBool("persist");
225 auto newChecksum = script.terminate().checksum();
226
227 auto ignoreScript = false;
228
229 if (!firstPacketReceived) {
230 auto isEqual = mChecksum == newChecksum;
231 ignoreScript = isEqual && mPersist && !mFailedWithError;
232 firstPacketReceived = true;
233 }
234
235 if (!ignoreScript) {
236 mChecksum = newChecksum;
237 mPersist = newPersist;
238 mFailedWithError = false;
239
240 // TODO: check signature here later
241 getLisp().runCode(script);
242 LispDevice::store();
243 } else {
244 UNIOT_LOG_INFO("script ignored: %s", script.c_str());
245 }
246
247 // Free memory to avoid fragmentation
248 object().clean();
249 // NOTE: you may need getLasCode() somewhere else, but it has already cleaned here
250 getLisp().cleanLastCode(); // Is it safe?
251 }
252
260 void handleEvent(const Bytes &payload) {
261 if (payload.size() > 0) {
262 // TODO: implement event validation
263 CoreEventListener::sendDataToChannel(unLisp::Channel::IN_EVENT, payload);
264 CoreEventListener::emitEvent(unLisp::Topic::IN_LISP_EVENT, unLisp::Msg::IN_NEW_EVENT);
265 }
266 }
