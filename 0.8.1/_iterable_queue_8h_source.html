115template <typename T>
116bool IterableQueue<T>::deleteCurrent() {
117 if (!mCurrent) {
118 return false;
119 }
120
121 if (mCurrent == ClearQueue<T>::mHead) {
122 ClearQueue<T>::hardPop();
123 mCurrent = ClearQueue<T>::mHead;
124 return true;
125 }
126
127 typename ClearQueue<T>::pnode prev = ClearQueue<T>::mHead;
128 while (prev && prev->next != mCurrent) {
129 prev = prev->next;
130 }
131
132 if (!prev) {
133 return false;
134 }
135
136 auto nextNode = mCurrent->next;
137 prev->next = nextNode;
138
139 if (!nextNode) {
140 ClearQueue<T>::mTail = prev;
141 }
142
143 delete mCurrent;
144 mCurrent = nextNode;
145
146 return true;
147}
