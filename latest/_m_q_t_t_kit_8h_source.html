101 MQTTKit(const Credentials &credentials, CBORExtender infoExtender = nullptr)
102 : mpCredentials(&credentials),
103 mPath(credentials),
104 mInfoExtender(infoExtender),
105 mPubSubClient(mWiFiClient),
106 mNetworkConnected(false),
107 mConnectionId(0) {
108 mPubSubClient.setCallback([this](char *topic, uint8_t *payload, unsigned int length) {
109 mDevices.forEach([&](MQTTDevice *device) {
110 if (device->isSubscribed(String(topic))) {
111 if (!length) {
112 device->handle(topic, Bytes());
113 return;
114 }
115
116 Bytes decoded;
117 if (_readCOSEMessage(Bytes(payload, length), decoded)) {
118 device->handle(topic, decoded);
119 } else {
120 UNIOT_LOG_ERROR("Failed to decode message on topic: %s", topic);
121 }
122 }
123 });
124 });

229 virtual void onEventReceived(unsigned int topic, int msg) override {
230 if (NetworkScheduler::CONNECTION == topic) {
231 switch (msg) {
232 case NetworkScheduler::SUCCESS:
233 mNetworkConnected = true;
234 Date::getInstance().forceSync();
235 break;
236 case NetworkScheduler::ACCESS_POINT:
237 case NetworkScheduler::CONNECTING:
238 case NetworkScheduler::DISCONNECTED:
239 case NetworkScheduler::FAILED:
240 default:
241 mNetworkConnected = false;
242 mTaskMQTT->detach();
243 break;
244 }
245 return;
246 }
247 if (Date::TIME == topic) {
248 switch (msg) {
249 case Date::SYNCED:
250 if (!mTaskMQTT->isAttached()) {
251 mTaskMQTT->attach(10);
252 }
253 break;
254 default:
255 break;
256 }
257 return;
258 }
259 }

274 inline void _initTasks() {
275 mTaskMQTT = TaskScheduler::make([this](SchedulerTask &self, short t) {
276 if (!mNetworkConnected) {
277 UNIOT_LOG_DEBUG("MQTT: Network is not connected");
278 return;
279 }
280 if (!mPubSubClient.connected()) {
281 UNIOT_LOG_DEBUG("Attempting MQTT connection #%d...", mConnectionId);
282 Bytes packetExtention;
283 if (mInfoExtender) {
284 CBORObject packet;
285 mInfoExtender(packet);
286 packetExtention = packet.build();
287 }
288
289 CBORObject offlineCBOR(packetExtention);
290 _prepareOfflinePacket(offlineCBOR);
291 auto offlinePacket = _buildCOSEMessage(offlineCBOR.build());
292 auto password = _getUserPassword();
293 if (mPubSubClient.connect(
294 _getClientId().c_str(),
295 _getUserLogin().c_str(),
296 (const char *)password.raw(),
297 password.size(),
298 mPath.buildDevicePath("status").c_str(),
299 0,
300 true,
301 (const char *)offlinePacket.raw(),
302 offlinePacket.size(),
303 true)) {
304 CBORObject onlineCBOR(packetExtention);
305 _prepareOnlinePacket(onlineCBOR);
306 auto onlinePacket = _buildCOSEMessage(onlineCBOR.build());
307 mPubSubClient.publish(
308 mPath.buildDevicePath("status").c_str(),
309 onlinePacket.raw(),
310 onlinePacket.size(),
311 true); // publish an announcement
312 mDevices.forEach([this](MQTTDevice *device) {
313 device->topics()->forEach([this](String topic) {
314 mPubSubClient.subscribe(topic.c_str());
315 });
316 });
317 CoreEventEmitter::emitEvent(Topic::CONNECTION, Msg::SUCCESS);
318 } else {
319 CoreEventEmitter::emitEvent(Topic::CONNECTION, Msg::FAILED);
320 }
321 }
322 mPubSubClient.loop();
323 });
324 }

332 Bytes _buildCOSEMessage(const Bytes &payload, bool sign = false) {
333 COSEMessage obj;
334 obj.setPayload(payload);
335 auto kid = mpCredentials->keyId(); // NOTE: dynamic data must be within the scope of the obj.build() function
336 if (sign) {
337 obj.sign(*mpCredentials);
338 obj.setUnprotectedKid(kid);
339 }
340 return obj.build();
341 }
342
350 bool _readCOSEMessage(const Bytes &message, Bytes &outPayload) {
351 COSEMessage obj(message);
352 if (obj.wasReadSuccessful()) {
353 outPayload = obj.getPayload();
354 return true;
355 }
356 return false;
357 }

409 Bytes _getUserPassword() {
410 CBORObject password;
411 auto protectedData = password.putMap("protected");
412 protectedData.put("device", mpCredentials->getDeviceId().c_str());
413 protectedData.put("owner", mpCredentials->getOwnerId().c_str());
414 protectedData.put("creator", mpCredentials->getCreatorId().c_str());
415 protectedData.put("timestamp", static_cast<int64_t>(Date::now()));
416 auto unprotectedData = password.putMap("unprotected");
417 unprotectedData.put("alg", "EdDSA");
418
419 auto signature = mpCredentials->sign(protectedData.build());
420 password.put("signature", signature.raw(), signature.size());
421
422 return password.build();
423 }
