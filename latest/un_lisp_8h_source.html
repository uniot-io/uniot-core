154 void runCode(const Bytes &data) {
155 if (!data.size())
156 return;
157
158 mLastCode = data;
159
160 mTaskLispEval->detach();
161 _destroyMachine();
162 _createMachine();
163
164 auto code = mLastCode.terminate().c_str();
165 UNIOT_LOG_DEBUG("eval: %s", code);
166
167 _refreshIncomingEvents();
168 lisp_eval(mLispRoot, mLispEnv, code);
169 if (!mTaskLispEval->isAttached()) {
170 _destroyMachine();
171 }
172 }

180 unLisp *pushPrimitive(Primitive *primitive) {
181 auto description = PrimitiveExpeditor::extractDescription(primitive);
182 mUserPrimitives.push(MakePair(description.name, primitive));
183 UNIOT_LOG_TRACE("primitive added: %s", description.name.c_str());
184 UNIOT_LOG_TRACE("args count: %d", description.argsCount);
185 UNIOT_LOG_TRACE("return type: %d", description.returnType);
186 for (auto i = 0; i < description.argsCount; i++) {
187 UNIOT_LOG_TRACE("arg %d: %d", i, description.argsTypes[i]);
188 }

289 mTaskLispEval = TaskScheduler::make([this](SchedulerTask &self, short t) {
290 auto root = mLispRoot;
291 auto env = mLispEnv;
292
293 DEFINE2(t_obj, t_pass);
294 *t_pass = get_variable(root, env, "#t_pass");
295 (*t_pass)->cdr->value = t;
296 *t_obj = get_variable(root, env, "#t_obj")->cdr;
297 safe_eval(root, env, t_obj);
298
299 if (!t) {

301 }

326 void _createMachine() {
327 lisp_create(UNIOT_LISP_HEAP);
328
329 *mLispEnv = make_env(mLispRoot, &Nil, &Nil);
330 define_constants(mLispRoot, mLispEnv);
331 define_primitives(mLispRoot, mLispEnv);
332
333 // Register special variables and built-in primitives
334 add_constant(mLispRoot, mLispEnv, "#t_obj", &Nil);
335 add_constant_int(mLispRoot, mLispEnv, "#t_pass", 0);
336 add_primitive(mLispRoot, mLispEnv, "task", mPrimitiveTask);
337 add_primitive(mLispRoot, mLispEnv, "is_event", mPrimitiveIsEventAvailable);
338 add_primitive(mLispRoot, mLispEnv, "pop_event", mPrimitivePopEvent);
339 add_primitive(mLispRoot, mLispEnv, "push_event", mPrimitivePushEvent);
340
341 // Register all user-defined primitives
342 mUserPrimitives.forEach([this](Pair<const String &, Primitive *> holder) {
343 add_primitive(mLispRoot, mLispEnv, holder.first.c_str(), holder.second);
344 });
345
346 UNIOT_LOG_DEBUG("lisp machine created, mem used: %d", lisp_mem_used());
347 }

369 void _pushIncomingEvent(const Bytes &eventData) {
370 constexpr size_t EVENTS_LIMIT = 5;
371
372 CBORObject event(eventData);
373 auto eventID = event.getString("eventID");
374 auto value = event.getValueAsString("value"); // NOTE: this is only used here to verify the correctness of the event
375
376 if (!eventID.isEmpty() && !value.isEmpty()) {
377 if (!mIncomingEvents.exist(eventID)) {
378 mIncomingEvents.put(eventID, std::make_shared<LimitedQueue<Bytes>>());
379 mIncomingEvents.get(eventID)->limit(EVENTS_LIMIT);
380 }
381 mIncomingEvents.get(eventID)->pushLimited(eventData);
382 }
383
384 // TODO: Find a way to remove events that are not consumed
385 }
386
394 bool _isIncomingEventAvailable(const String &eventID) {
395 return mIncomingEvents.exist(eventID) && mIncomingEvents.get(eventID)->size();
396 }
397
405 Bytes _popIncomingEvent(const String &eventID) {
406 if (_isIncomingEventAvailable(eventID)) {
407 return mIncomingEvents.get(eventID)->popLimited({});
408 }
409 return {};
410 }

420 bool _pushOutgoingEvent(String eventID, int value) {
421 CBORObject event;
422 event.put("eventID", eventID.c_str());
423 event.put("value", value);
424
425 auto sent = this->sendDataToChannel(unLisp::Channel::OUT_EVENT, event.build());
426 this->emitEvent(Topic::OUT_LISP_EVENT, Msg::OUT_NEW_EVENT);
427 return sent;
428 }

441 inline Object _primTask(Root root, VarObject env, VarObject list) {
442 auto expeditor = PrimitiveExpeditor::describe("task", Lisp::Bool, 3, Lisp::Int, Lisp::Int, Lisp::Cell)
443 .init(root, env, list);
444 expeditor.assertDescribedArgs();
445
446 auto times = expeditor.getArgInt(0);
447 auto ms = expeditor.getArgInt(1);
448 auto obj = expeditor.getArg(2);
449
450 DEFINE1(t_obj);
451 *t_obj = get_variable(root, env, "#t_obj");
452 (*t_obj)->cdr = obj;
453
454 mTaskLispEval->attach(ms, times);
455
456 return expeditor.makeBool(true);
457 }

492 inline Object _primPopEvent(Root root, VarObject env, VarObject list) {
493 auto expeditor = PrimitiveExpeditor::describe("pop_event", Lisp::Bool, 1, Lisp::Symbol)
494 .init(root, env, list);
495 expeditor.assertDescribedArgs();
496
497 auto eventId = expeditor.getArgSymbol(0);
498 auto eventData = _popIncomingEvent(eventId);
499 CBORObject event(eventData);
500 auto value = event.getValueAsString("value");
501
502 auto number = value.toInt();
503 auto isNumber = number || value == "0";
504
505 UNIOT_LOG_WARN_IF(!isNumber, "event value is not a number");
506
507 return expeditor.makeInt(number);
508 }

521 inline Object _primPushEvent(Root root, VarObject env, VarObject list) {
522 auto expeditor = PrimitiveExpeditor::describe("push_event", Lisp::Bool, 2, Lisp::Symbol, Lisp::BoolInt)
523 .init(root, env, list);
524 expeditor.assertDescribedArgs();
525
526 auto eventId = expeditor.getArgSymbol(0);
527 auto value = expeditor.getArgInt(1);
528
529 auto sent = _pushOutgoingEvent(eventId, value);
530
531 return expeditor.makeBool(sent);
532 }
