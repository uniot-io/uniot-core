360 inline void _initListeners() {
361 mpNetworkEventListener = MakeUnique<CoreCallbackEventListener>([&](int topic, int msg) {
362 if (NetworkScheduler::CONNECTION == topic) {
363 switch (msg) {
364 case NetworkScheduler::SUCCESS:
365 UNIOT_LOG_DEBUG("AppKit Subscriber, SUCCESS, ip: %s", WiFi.localIP().toString().c_str());
366 break;
367 case NetworkScheduler::ACCESS_POINT:
368 UNIOT_LOG_DEBUG("AppKit Subscriber, ACCESS_POINT");
369 mpNetworkEventListener->receiveDataFromChannel(NetworkScheduler::Channel::OUT_SSID, [this](unsigned int id, bool empty, Bytes data) {
370 if (!empty) {
371 UNIOT_LOG_DEBUG("SSID: %s", data.terminate().c_str());
372 }
373 });
374 break;
375
376 case NetworkScheduler::CONNECTING:
377 UNIOT_LOG_DEBUG("AppKit Subscriber, CONNECTING");
378 mpNetworkEventListener->receiveDataFromChannel(NetworkScheduler::Channel::OUT_SSID, [this](unsigned int id, bool empty, Bytes data) {
379 if (!empty) {
380 UNIOT_LOG_DEBUG("SSID: %s", data.terminate().c_str());
381 }
382 });
383 break;
384
385 case NetworkScheduler::DISCONNECTED:
386 UNIOT_LOG_DEBUG("AppKit Subscriber, DISCONNECTED");
387 break;
388
389 case NetworkScheduler::FAILED:
390 default:
391 UNIOT_LOG_DEBUG("AppKit Subscriber, FAILED");
392 break;
393 }
394 return;
395 }
396 if (MQTTKit::CONNECTION == topic) {
397 switch (msg) {
398 case MQTTKit::SUCCESS:
399 UNIOT_LOG_DEBUG("AppKit Subscriber, MQTT SUCCESS");
400 mMQTT.renewSubscriptions();
401 break;
402 case MQTTKit::FAILED:
403 default:
404 UNIOT_LOG_DEBUG("AppKit Subscriber, MQTT FAILED");
405 break;
406 }
407 return;
408 }
409 });
410 }
